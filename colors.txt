acc = 3
function iff(args)
  args.focusable = args.focusable or false
  args.c = (args.c or 0) + 50
  if not (args.w or args.text) then
    args.w = 30
  end
  if not args.r then
    args.r = acc
    acc = acc + args.h
  end
  group[b.f(args)] = true
end
